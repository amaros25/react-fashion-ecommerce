    import MapPickerPopup from "../maps/map_picker_popup";

    import { MapContainer, TileLayer, Marker, Popup } from "react-leaflet";
    import L from "leaflet";

    const [position, setPosition] = useState(null);
    const [tempPosition, setTempPosition] = useState(null);
    const [showMapPopup, setShowMapPopup] = useState(false);

     <button
        onClick={() => {
            if (navigator.geolocation) {
            navigator.geolocation.getCurrentPosition(
                (pos) => {
                const { latitude, longitude } = pos.coords;
                setShowMapPopup(true);
                setTempPosition([latitude, longitude]);
                },
                () => alert(t("cart_page.gps_denied"))
            );
            } else {
            alert(t("cart_page.geolocation_unsupported"));
            }
        }}

        >

        
        üìç {t("cart_page.use_my_position")}
        </button>
        {showMapPopup && (
        <MapPickerPopup
            initialPosition={tempPosition || [51.505, -0.09]} // fallback Position
            onClose={(selectedPosition) => {
            setShowMapPopup(false);
            if (selectedPosition) setPosition(selectedPosition);
            }}
        />
        )}
        {position && (
            <MapContainer
                center={position}
                zoom={13}
                style={{ height: "300px", width: "100%", borderRadius: "10px", marginTop: "1rem" }}
            >
                <TileLayer
                url="https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png"
                attribution='&copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a> contributors'
                />
                <Marker position={position}>
                <Popup>üìç {t("cart_page.current_position")}</Popup>
                </Marker>
            </MapContainer>
            )}
